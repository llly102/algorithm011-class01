class Solution {
    public int search(int[] nums, int target) {
        if(nums.length==0){
            return -1;
        }
        int left = 0 , right = nums.length-1;
        int mid;
        while(left<=right){
            mid=(left+right)/2;
            if(nums[mid] == target){
                return mid;
            }
            //前半部分有序
            if(nums[mid]>=nums[left]){
                //target在前半部分中
                if(target>=nums[left] && target<nums[mid]){
                    right = mid-1;
                }
                else left = mid+1;
            }
            //后半部分有序
            else {
                //target在后半部分中
                if(target > nums[mid] && target <= nums[right]){
                    left = mid+1;
                }
                else right = mid-1;
            }
        }
        return -1;
    }
}
