class Solution {
    public int findContentChildren(int[] g, int[] s) {
        if(g.length==0 || s.length==0) {
            return 0;
        }
        Arrays.sort(g);
        Arrays.sort(s);
        int count = 0;

        int i=0,j=0;
        while(i<g.length && j<s.length){
            if(g[i]<=s[j]){
                count++;
                i++;
                j++;
            }
            else j++;
        }
        return count;

    }
}
